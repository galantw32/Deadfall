function startTimer() {
  state.timerInterval = setInterval(() => {
    state.timer--;
    const mins = Math.floor(state.timer / 60).toString().padStart(2,'0');
    const secs = (state.timer % 60).toString().padStart(2,'0');
    const el = document.getElementById('timer');
    el.textContent = `${mins}:${secs}`;
    if (state.timer <= 60) el.classList.add('urgent');
    if (state.timer <= 0) {
      clearInterval(state.timerInterval);
      forceDecision();
    }
  }, 1000);
}

let currentAdvisorId = null;

function openBriefing(advisorId) {
  const adv = ADVISORS.find(a => a.id === advisorId);
  const idx = state.briefingIndex[advisorId] || 0;
  
  if (idx >= adv.briefings.length) {
    addIntel('SYSTEM', `${adv.name} has no further intelligence to provide.`, '', 'tag-amber', 'EXHAUSTED');
    return;
  }

  currentAdvisorId = advisorId;
  const briefing = adv.briefings[idx];

  document.getElementById('b-name').textContent = adv.name;
  document.getElementById('b-title').textContent = adv.title + ' — CLASSIFIED BRIEFING';
  document.getElementById('b-text').textContent = `"${briefing.text}"`;
  document.getElementById('b-rec').textContent = briefing.rec;

  document.getElementById('briefing-modal').classList.add('show');
}

function closeBriefing() {
  document.getElementById('briefing-modal').classList.remove('show');
  currentAdvisorId = null;
}

function trustAdvisor() {
  if (!currentAdvisorId) return;
  const adv = ADVISORS.find(a => a.id === currentAdvisorId);
  const idx = state.briefingIndex[currentAdvisorId] || 0;
  const briefing = adv.briefings[idx];

  // Add evidence
  if (briefing.evidenceAdded && !state.evidenceList.includes(briefing.evidenceAdded)) {
    state.evidenceList.push(briefing.evidenceAdded);
    renderEvidence();
  }

  // Auto-select target if advisor has one and it's trusted
  if (briefing.target) {
    const suspect = SUSPECTS.find(s => s.name === briefing.target);
    if (suspect) {
      addIntel(adv.name.split(' ').pop(), `Intel accepted. Updating threat assessment — ${briefing.target} probability elevated.`, 'conflicting', 'tag-amber', 'INTEL ACCEPTED');
    }
  } else {
    addIntel(adv.name.split(' ').pop(), `Intel accepted. Strike recommendation: WITHHOLD. Further verification counseled.`, '', 'tag-green', 'CAUTION');
  }

  // Advance briefing index
  state.briefingIndex[currentAdvisorId] = idx + 1;
  if (!state.trustedAdvisors.includes(currentAdvisorId)) state.trustedAdvisors.push(currentAdvisorId);

  renderAdvisors();
  renderTrustDisplay();
  closeBriefing();
}

function dismissAdvisor() {
  if (!currentAdvisorId) return;
  const adv = ADVISORS.find(a => a.id === currentAdvisorId);
  const idx = state.briefingIndex[currentAdvisorId] || 0;

  addIntel('SYSTEM', `Intel from ${adv.name} disregarded by presidential order.`, '', 'tag-red', 'DISREGARDED');
  state.briefingIndex[currentAdvisorId] = idx + 1;
  if (!state.dismissedAdvisors.includes(currentAdvisorId)) state.dismissedAdvisors.push(currentAdvisorId);
  
  renderAdvisors();
  closeBriefing();
}

      details = `OUTCOME: WRONG TARGET STRUCK\n${target.name} retaliates through proxy channels.\nNSA forensic analysis: all intelligence pointing to ${target.name} was fabricated.\nThe war that follows will not be won.`;
    }
  }

  setTimeout(() => showOutcome(outcome, guilty, body, details), 800);
}

function showOutcome(title, success, body, details) {
  const el = document.getElementById('outcome');
  document.getElementById('outcome-title').textContent = title;
  document.getElementById('outcome-title').className = 'outcome-title ' + (success ? 'success' : 'failure');
  document.getElementById('outcome-body').textContent = body;
  document.getElementById('outcome-details').textContent = details;
  el.classList.add('show');
}
